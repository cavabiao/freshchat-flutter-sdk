int priorityValue() {
  switch (this) {
    case Priority.PRIORITY_DEFAULT:
      return 0;
      break;

    case Priority.PRIORITY_LOW:
      return -1;
      break;

    case Priority.PRIORITY_MIN:
      return -2;
      break;

    case Priority.PRIORITY_HIGH:
      return 1;
      break;

    case Priority.PRIORITY_MAX:
      return 2;
      break;

    default:
      return 0;
      break;
  }
}
